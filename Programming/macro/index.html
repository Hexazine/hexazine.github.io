マクロ

競プロメモの最後に載せるコードがお手本ではない(あくまで参考である)癖に冗長になるのが嫌なので、LLとかrepとかのマクロは基本的に使用します。コードはパクらずに自分で書こうね！
ただこれのせいで可読性が落ちまくってる可能性があるので、流石に使わなそうなやつ含めて今僕が使っているマクロ全部載せておきます。 
#define LL long long
#define DB double
#define LD long double
#define ST string
#define BS bitset
#define PA pair<LL,LL>
#define VE vector
#define VL VE<LL>
#define VP VE<PA>
#define VVL VE<VL>
#define VVVL VE<VVL>
#define PQ priority_queue
#define FI first
#define SE second
#define PB push_back
#define POB pop_back
#define PF push_front
#define POF pop_front
#define MP make_pair
#define rep1(i,a,b,c) for(int i=a;i<b;i+=c)
#define rep2(i,a,b) rep1(i,a,b,1)
#define rep3(i,a) rep2(i,0,a)
#define rep4(a) rep3(i,a)
#define repr1(i,a,b,c) for(int i=b-1;i>=a;i-=c)
#define repr2(i,a,b) repr1(i,a,b,1)
#define repr3(i,a) repr2(i,0,a)
#define repr4(a) repr3(i,a)
#define overload4(a,b,c,d,e,...) e
#define rep(...) overload4(__VA_ARGS__,rep1,rep2,rep3,rep4)(__VA_ARGS__)
#define repr(...) overload4(__VA_ARGS__,repr1,repr2,repr3,repr4)(__VA_ARGS__)
#define ALL(a) a.begin(),a.end()
#define RALL(a) a.rbegin(),a.rend()
#define SORT(a) sort(ALL(a))
#define REV(a) reverse(ALL(a))
#define UB(a,n) *upper_bound(ALL(a),n)
#define UBn(a,n) upper_bound(ALL(a),n)-a.begin()
#define LB(a,n) *lower_bound(ALL(a),n)
#define LBn(a,n) lower_bound(ALL(a),n)-a.begin()
#define INF 1000000000000000003
#define PI 3.14159265358979323846264338327950288
#define MOD 998244353
#define EPS 1e-10
